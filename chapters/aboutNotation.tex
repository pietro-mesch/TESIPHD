\section*{About Notation}
A quick glossary of the relevant variables is provided below, alongside the units of each dimensional quantity.

For a leaner presentation of the model, subscripts referring to
topological elements may be dropped to simplify notation.

\subsubsection*{Network Topology}
\begin{tabu} to \textwidth {X[3,c] X[1,c] X[6,l]}
$i,j \in \nodset$ & & nodes (junctions) \\[2pt]

$a,b \in \arcset \subseteq \nodset \times \nodset $ & & arcs (lane groups) \\[2pt]

$\length_a$ & \units{m} & length of arc $a$ \\[2pt]

$\left( \tail{a} \; , \;\head{a} \right) = a$ & & tail and head nodes of arc $a$ \\[2pt]

$\fstar{i} = \left\lbrace a \in \arcset \, | \, \nodset_{a}^{-} = i \right\rbrace $ & & forward star of node $i$ (outgoing arcs)\\[2pt]

$\bstar{i} = \left\lbrace a \in \arcset \, | \, \nodset_{a}^{+} = i \right\rbrace $ & & backward star of node $i$ (incoming arcs)\\[2pt]

$y,z \in \manset$ & & manoeuvres\\

\end{tabu} 


\subsubsection*{Signal Phases}
\begin{tabu} to \textwidth {X[3,c] X[1,c] X[6,l]}
$p,q \in \phaset_{j}$ & & signal phases at junction $j$\\[2pt]

$\arcset_{p} \subseteq \bstar{j} $ & & lane groups open during phase $p$ \\[2pt]

\end{tabu} 


\subsubsection*{Signal Timing}
\begin{tabu} to \textwidth {X[3,c] X[1,c] X[6,l]}
$ \cycle{j} $ & \units{s} & cycle time at junction $j$\\[2pt]
$ \tlost{j} $ & \units{s} & time lost per cycle at junction $j$\\[2pt]
$ \toffs{j} $ & \units{s} & offset of junction $j$ \\[5pt]
$ t_p   $ & \units{s} & nominal duration of phase $p$\\[2pt]
$ \gini{a}   $ &  & initial instant of the green phase for arc $a$\\[2pt]
$ \gren{a}   $ & \units{s} & effective green duration for arc $a$\\[3pt]
$ \gint{a} = \left[\gini{a} \, , \gini{a}+\gren{a} \right] $ &  & span of the green phase for arc $a$\\[4pt]
$ \displaystyle \gshr{a} = \frac{\gren{a}}{t_j^C} $ & \units{\%} & effective green share of arc $a$\\[2pt]
\end{tabu} 


\subsubsection*{Demand and Supply}
\begin{tabu} to \textwidth {X[3,c] X[1,c] X[6,l]}
$ \flow_{a} $ & \units{veh/s} & demand flow on arc $a$ \\[2pt]
$ \satflow_a $ & \units{veh/s} & saturation flow of arc $a$ \\[2pt]

$ \displaystyle \flowratio_a = \frac{\flow_a}{\satflow_a} $ & & flow ratio on arc $a$ \\[2pt]
$ \displaystyle \saturation_a = \frac{\flowratio_a}{\gamma_a} $ & & saturation on arc $a$ \\[2pt]
\end{tabu} 

\subsubsection*{Simulation Parameters}
\begin{tabu} to \textwidth {X[3,c] X[1,c] X[6,l]}
$ \tau \in \simspan $ &  & intervals of the simulation window \\[2pt]
$ \simintd $ & s & duration of interval $\tau$\\[2pt]
\end{tabu}

\subsubsection*{Simulation Results}
\begin{tabu} to \textwidth {X[3,c] X[1,c] X[6,l]}
$ \fcum{a,t} $ & \units{veh} & vehicles that entered arc $a$ before time $t$\\[2pt]
$ \gcum{a,t} $ & \units{veh} & spaces that reached $\tail{a}$ before time $t$\\[2pt]
$ \hcum{a,t} $ & \units{veh} & vehicles that reached $\head{a}$ before time $t$\\[2pt]
$ \ecum{a,t} $ & \units{veh} & vehicles that left arc $a$ before time $t$\\[2pt]
$ \nveh{a,t} $ & \units{veh} & vehicles on arc $a$ at time $t$\\[2pt]
$ \traveltime{a,t} $ & \units{s} & travel time for arc $a$ entering at time $t$ \\[2pt]
$ \quen{a,t} $ & \units{veh} & vehicles in the vertical queue at $\head{a}$ at time $t$\\[2pt]

\end{tabu}

\subsubsection*{Performance Indicators}
\begin{tabu} to \textwidth {X[3,c] X[1,c] X[6,l]}
$ \tclear{a} $ & s & queue clearance time on arc $a$ (per cycle)\\[2pt]
$\queuerelative{a}$ & \units{\%} & queue length relative to total length of arc $a$ \\[2pt]
$ \avgdelay{a} $ & s & average delay of arc $a$ \\[2pt]
$ \stopratio{a}$ & & share of $\flow_a$ stopping at or before $\nodset_{a}^{+}$ \\[2pt]
$ \total{a} \text{, } \total{\corridor} $ & \units{veh} & total inflow to arc $a$ or all sections of corridor $\corridor$ \\[2pt]
$ \utspent{a} \text{, } \utspent{\corridor} $ & \units{s} & user time spent on arc $a$ or corridor $\corridor$ \\[2pt]
\end{tabu}
